import os

dir_c = {}

for item in os.listdir("."):
    if os.path.isdir(item):
        files = os.listdir(item)
        dir_c[item] = len(files)
        
max_dir = max(dir_c, key=dir_c.get)
print(f"eng ko;p faylli papka:{max_dir}, fayllar soni:{dir_c[max_dir]}")

import json
from datetime import datetime
def decorator(func):
    def wrapper(*args, **kwargs):
        function_name = {
                "name": func.__name__,
                "call_time": datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
            }
        with open("function_calls.json", "w") as f:
            json.dump(function_name, f, indent=4)
        return func(*args, **kwargs)
    return wrapper

@decorator
def func1():
    print("func1")
@decorator
def func2():
    print("func2")

func1()
func2()

class ATM:
    def __init__(self,balance=0):
        self.balance = balance
        with open("balance.txt","w") as f:
            f.write(str(self.balance))

    def get_balance(self):
        with open("balance.txt") as f:
            balance = int(f.read())
            print(balance)

    def deposit(self,amount):
        self.balance += amount
        with open("balance.txt","w") as f:
            f.write(str(self.balance))

    def withdraw(self,amount):
        self.balance -= amount
        with open("balance.txt","w") as f:
            f.write(str(self.balance))

user = ATM(balance=100)
user.deposit(100)
user.withdraw(100)
